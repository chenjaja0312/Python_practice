"""最大公因數"""

a,b=eval(input("請輸入正整數a,b:"))

#求最大公因數、一剛開始先設目前最大公因數=0
gcd=0
# 從2開始:找「最大」，1 沒意義
# min: 公因數不可能大於較小的那個數
#為了找是誰為公因數，所以必須用迴圈一個一個找

for k in range(2,min(a,b)+1):
# 這邊意思 是否a,b整除
  if a%k == 0 and b%k == 0:
    if k>gcd:
      gcd = k
if gcd != 0:
  print("最大公因數為:",gcd)
else:
  print("兩數互為質數")

# Palindrome 回文
def isPalindrome(s):

  #把「所有空格」拿掉、全部轉成小寫
  s=s.replace(" ","").lower()

  #原本的字串跟反過來的字串是不是一樣？」
  return s==s[::-1]

print(isPalindrome("racecar"))
print(isPalindrome("Hello"))
print(isPalindrome("A man a plan a canal Panama"))

#這邊 除了power 其他打包在一起
def powerSum(power, *args):
    total = 0
    for num in args:
      #算 num 的power 次方
      total += num ** power
    return total

# Calculate 1^2 + 2^2 + 3^2 + 4^2
# 2,(1,2,3,4)
print(powerSum(2, 1, 2, 3, 4))

# Calculate 1^3 + 2^3 + 3^3
# 3,(1,2,3)
print(powerSum(3, 1, 2, 3))

# Calculate 1^4 + 2^4
# 4,(1,2)
print(powerSum(4, 1, 2))
